\newcommand{\ScoreField}[1]{%
  \Label{#1}\hspace{1.5mm}\GuideRule{16mm}\,\textcolor{guide}{/10}%
}

\newcommand{\ScoreRowThree}[3]{%
  \noindent\begin{tabular*}{\linewidth}{@{}p{0.33\linewidth}p{0.33\linewidth}p{0.33\linewidth}@{}}%
    \ScoreField{#1} & \ScoreField{#2} & \ScoreField{#3} \\
  \end{tabular*}\par
  \vspace{2.6mm}
}

\newcommand{\ScoreRowTwo}[2]{%
  \noindent\begin{tabular*}{\linewidth}{@{}p{0.5\linewidth}p{0.5\linewidth}@{}}%
    \ScoreField{#1} & \ScoreField{#2} \\
  \end{tabular*}\par
  \vspace{2.6mm}
}

\newcommand{\WriteLines}[2][6.8mm]{%
  \vspace{0.6mm}\GuideLine
  \ifnum#2>1
    \foreach \i in {2,...,#2}{%
      \vspace{#1}\GuideLine
    }%
  \fi
  \vspace{1.6mm}
}

\newcommand{\Separator}{%
  \vspace{1.2mm}
  \begin{center}
    \begin{tikzpicture}[x=1mm,y=1mm]
      \draw[color=guide, line width=0.2pt]
        (0,0)
        .. controls (3,0.7) and (6,-0.7) .. (9,0)
        .. controls (12,0.7) and (15,-0.7) .. (18,0);
    \end{tikzpicture}
  \end{center}
  \vspace{1.6mm}
}

\newcommand{\WriteNumberedLines}[2][10.5mm]{%
  \foreach \i in {1,...,#2}{%
    \vspace{#1}%
    \noindent\makebox[6mm][l]{\textcolor{guide}{\small \i}}%
    \GuideRule{\dimexpr\linewidth-6mm\relax}\par
  }%
  \vspace{1.6mm}
}

% ---- page types ----
\newcommand{\MorningPage}[1]{%
  \DateLineSection{#1}{Morning}
  \ScoreRowThree{Sleep}{Energy}{Mood}
  \Separator

  \Prompt{Grateful for:}
  \WriteNumberedLines{3}
  \vfill

  \Prompt{Today will be a win if:}
  \WriteLines{2}
  \vfill

  \Prompt{One priority:}
  \WriteLines{2}
  \vfill

  \Prompt{How I'll show up today:}
  \WriteLines{2}
  \vspace{4mm}
}

\newcommand{\EveningPage}[1]{%
  \DateLineSection{#1}{Evening}
  \ScoreRowTwo{Mood}{Energy}
  \Separator

  \Prompt{3 good things that happened:}
  \WriteNumberedLines{3}
  \vfill

  \Prompt{Why they happened / what I did right:}
  \WriteLines{3}
  \vfill

  \Prompt{One adjustment for tomorrow:}
  \WriteLines{3}
  \vspace{4mm}
}

\newcommand{\WeeklyPage}[1]{%
  {\LARGE \textbf{Week of #1}}\par
  \vspace{1mm}
  \GuideLine
  \vspace{3mm}

  \Prompt{3 highlights:}
  \WriteLines{3}
  \Separator

  \Prompt{What worked:}
  \WriteLines{2}
  \Separator

  \Prompt{One improvement:}
  \WriteLines{2}
  \Separator

  \Prompt{Intention for next week:}
  \WriteLines{1}
}
